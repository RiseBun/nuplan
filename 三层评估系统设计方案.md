### 2.3 核心接口设计

```python
class BaseWorldModel(ABC):
    """World Model 抽象基类"""
    
    @abstractmethod
    def generate(
        self,
        history_images: torch.Tensor,      # (B, T_h, C, H, W)
        ego_state: torch.Tensor,            # (B, ego_dim)
        candidate_actions: torch.Tensor,    # (B, T_f, action_dim)
        num_samples: int = 1,
        temperature: float = 1.0,
    ) -> Dict[str, torch.Tensor]:
        """
        生成未来图像
        
        返回:
        {
            'generated_images': (B, num_samples, T_f, C, H, W),
            'confidence': (B, num_samples),  # 生成置信度
        }
        """
        pass
    
    @abstractmethod
    def load_checkpoint(self, checkpoint_path: str):
        pass
```

### 3.2 FID 计算

```python
class FIDCalculator:
    """Fréchet Inception Distance"""
    
    def __init__(self, device='cuda'):
        self.inception = load_inception_v3()
        self.device = device
    
    def compute(
        self,
        generated_images: torch.Tensor,  # (N, C, H, W)
        real_images: torch.Tensor,       # (M, C, H, W)
    ) -> float:
        """
        计算 FID
        
        返回: FID score (越低越好)
        """
        # 提取 Inception 特征
        gen_features = self._extract_features(generated_images)
        real_features = self._extract_features(real_images)
        
        # 计算 FID
        fid = self._calculate_fid(gen_features, real_features)
        return fid
```

class FVDCalculator:

### 3.4 综合评估脚本

```python
# metrics/generation_eval.py

class GenerationEvaluator:
    """生成质量综合评估"""
    
    def __init__(self, config):
        self.fid_calc = FIDCalculator()
        self.fvd_calc = FVDCalculator()
        self.lpips_calc = LPIPSCalculator()
    
    def evaluate(
        self,
        generated_futures: torch.Tensor,
        ground_truth_futures: torch.Tensor,
    ) -> Dict[str, float]:
        """
        全面评估生成质量
        
        返回:
        {
            'fid': float,
            'fvd': float,
            'lpips': float,
            'overall_generation_score': float,
        }
        """
        # 计算各项指标
        fid = self.fid_calc.compute(generated_futures, ground_truth_futures)
        fvd = self.fvd_calc.compute(generated_futures, ground_truth_futures)
        lpips = self.lpips_calc.compute(generated_futures, ground_truth_futures)
        
        # 综合评分（归一化后加权）
        overall_score = self._compute_overall_score(fid, fvd, lpips)
        
        return {
            'fid': fid,
            'fvd': fvd,
            'lpips': lpips,
            'overall_generation_score': overall_score,
        }
```

### 4.2 Critic-Guided Planner

```python
class CriticGuidedPlanner:
    """使用 Critic 评分指导规划"""
    
    def __init__(self, world_model, critic, config):
        self.world_model = world_model
        self.critic = critic
        self.config = config
    
    def plan(
        self,
        history_images: torch.Tensor,
        ego_state: torch.Tensor,
        candidate_actions: torch.Tensor,
    ) -> Dict[str, Any]:
        """
        生成并选择最佳规划
        
        流程:
        1. World Model 生成多个候选未来
        2. Critic 对每个候选打分
        3. 选择最高分的候选
        4. 提取对应的规划轨迹
        """
        # Step 1: 生成候选未来
        generated_futures = self.world_model.generate(
            history_images, ego_state, candidate_actions,
            num_samples=self.config.num_samples_per_action
        )
        
        # Step 2: Critic 打分
        scores = []
        for i, future in enumerate(generated_futures):
            score = self.critic.score(
                history_images=history_images,
                future_images=future,
                ego_state=ego_state,
                candidate_action=candidate_actions[i]
            )
            scores.append(score)
        
        # Step 3: 选择最佳
        best_idx = torch.argmax(scores)
        best_action = candidate_actions[best_idx]
        
        return {
            'selected_action': best_action,
            'scores': scores,
            'best_score': scores[best_idx],
            'all_generated_futures': generated_futures,
        }
```

### 4.3 闭环评估流程

```python
class ClosedLoopEvaluator:
    """闭环性能评估"""
    
    def __init__(self, simulator, planner, config):
        self.simulator = simulator  # nuPlan simulator
        self.planner = planner
        self.config = config
    
    def evaluate_scenario(
        self,
        scenario: NuPlanScenario,
    ) -> Dict[str, float]:
        """
        在单个场景上运行闭环评估
        
        返回:
        {
            'pdm_score': float,
            'collision_rate': float,
            'off_road_rate': float,
            'comfort_score': float,
            'progress': float,
        }
        """
        # 在 simulator 中运行
        result = self.simulator.run(
            scenario=scenario,
            planner=self.planner,
            num_steps=self.config.max_steps
        )
        
        # 提取指标
        metrics = self._extract_metrics(result)
        return metrics
    
    def evaluate_dataset(
        self,
        scenarios: List[NuPlanScenario],
    ) -> Dict[str, float]:
        """在数据集上评估"""
        all_metrics = []
        for scenario in tqdm(scenarios):
            metrics = self.evaluate_scenario(scenario)
            all_metrics.append(metrics)
        
        # 聚合统计
        return self._aggregate_metrics(all_metrics)
```

### 4.4 相关性分析

```python
class CorrelationAnalyzer:
    """分析评估指标与闭环性能的相关性"""
    
    def analyze(
        self,
        scenarios: List[NuPlanScenario],
        planners: List[BasePlanner],
    ) -> Dict[str, Any]:
        """
        核心实验: 证明 critic score 与 closed-loop performance 相关性 > FID/FVD
        """
        results = {
            'fid_vs_closed_loop': [],
            'fvd_vs_closed_loop': [],
            'critic_vs_closed_loop': [],
        }
        
        for scenario in tqdm(scenarios):
            for planner in planners:
                # 运行闭环
                closed_loop_result = self.run_closed_loop(scenario, planner)
                pdm_score = closed_loop_result['pdm_score']
                
                # 获取各种评估分数
                fid_score = closed_loop_result['generation_fid']
                fvd_score = closed_loop_result['generation_fvd']
                critic_score = closed_loop_result['critic_score']
                
                results['fid_vs_closed_loop'].append((fid_score, pdm_score))
                results['fvd_vs_closed_loop'].append((fvd_score, pdm_score))
                results['critic_vs_closed_loop'].append((critic_score, pdm_score))
        
        # 计算相关性
        correlations = {}
        for metric_name, pairs in results.items():
            scores = [p[0] for p in pairs]
            pdm_scores = [p[1] for p in pairs]
            
            correlations[metric_name] = {
                'spearman': spearmanr(scores, pdm_scores),
                'pearson': pearsonr(scores, pdm_scores),
                'kendall': kendalltau(scores, pdm_scores),
            }
        
        # 验证假设: critic correlation > FID/FVD correlation
        hypothesis_test = self._test_hypothesis(correlations)
        
        return {
            'correlations': correlations,
            'hypothesis_test': hypothesis_test,
            'conclusion': self._draw_conclusion(correlations),
        }
```

- FID/FVD 可以离线预计算
